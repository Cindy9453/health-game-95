<script>
    const canvas = document.getElementById('game-canvas');
    const ctx = canvas.getContext('2d');

    // 遊戲資料與變數 (保留原本邏輯)
    let quizBank = [
        { q: "成人每日建議飲水量是體重(kg)乘以多少？", options: ["10ml", "20ml", "30-35ml", "100ml"], correct: 2 },
        { q: "下列哪種運動最能增進心肺功能？", options: ["看電視", "慢跑", "睡覺", "滑手機"], correct: 1 },
        { q: "洗手「內外夾弓大立腕」，最後動作是？", options: ["指尖", "手背", "手心", "手腕"], correct: 3 },
        { q: "預防近視，用眼30分鐘應休息多久？", options: ["1分鐘", "10分鐘", "30分鐘", "不必"], correct: 1 },
        { q: "正確刷牙法「貝氏刷牙法」，刷毛與牙齒夾角？", options: ["15度", "45度", "90度", "180度"], correct: 1 },
        { q: "下列哪種食物含有豐富的維生素C？", options: ["白米飯", "芭樂", "炸雞", "可樂"], correct: 1 },
        { q: "吃太鹹容易引發哪種疾病？", options: ["高血壓", "近視", "香港腳", "感冒"], correct: 0 },
        { q: "BMI指數是用來測量什麼？", options: ["智力", "體重是否標準", "視力", "聽力"], correct: 1 },
        { q: "哪種飲料最健康？", options: ["珍珠奶茶", "純水", "能量飲", "碳酸飲料"], correct: 1 },
        { q: "均衡飲食中，哪類食物應吃最多？", options: ["油脂類", "全穀雜糧類", "糖果類", "炸雞類"], correct: 1 }
    ];

    let currentLevel = 0;
    let score = 0;
    let hp = 3;
    let doorOpen = false;
    let gameStarted = false;
    let isWaiting = false;
    let animationId = null;
    let platforms = [];
    let coins = [];
    let wrongAnswers = [];
    let bridgeDir = 1;
    
    // 背景星星粒子
    let stars = [];
    for(let i=0; i<100; i++) {
        stars.push({ x: Math.random()*800, y: Math.random()*520, size: Math.random()*2, speed: Math.random()*0.5 });
    }

    const player = {
        x: 50, y: 50, vx: 0, vy: 0, size: 24,
        onGround: false, speed: 3.5, jumpPower: -10, gravity: 0.42
    };

    const keys = {};

    function startGame() {
        if (gameStarted) return;
        document.getElementById('start-screen').style.display = 'none';
        quizBank.sort(() => Math.random() - 0.5);
        gameStarted = true;
        initLevel();
        if (animationId) cancelAnimationFrame(animationId);
        gameLoop();
    }

    function generateMap() {
        platforms = [
            { x: 0, y: 130, w: 350, h: 15, isMovable: false },
            { x: 120, y: 280, w: 450, h: 15, isMovable: currentLevel >= 10, minX: 50, maxX: 300 },
            { x: 0, y: 430, w: 800, h: 20, isMovable: false }
        ];
        coins = [];
        const data = quizBank[currentLevel];
        const possiblePositions = [
            {pIdx: 1, xRange: [50, 150]},
            {pIdx: 1, xRange: [250, 350]},
            {pIdx: 2, xRange: [200, 350]},
            {pIdx: 2, xRange: [400, 600]}
        ].sort(() => Math.random() - 0.5);

        data.options.forEach((opt, i) => {
            const pos = possiblePositions[i];
            const p = platforms[pos.pIdx];
            const offsetX = pos.xRange[0] + Math.random() * (pos.xRange[1] - pos.xRange[0]);
            coins.push({ 
                pIdx: pos.pIdx, 
                offsetX: offsetX, 
                x: p.x + offsetX, 
                y: p.y - 45, 
                text: opt, 
                isCorrect: i === data.correct, 
                collected: false 
            });
        });
        player.x = 50; player.y = 50; player.vx = 0; player.vy = 0;
    }

    function initLevel() {
        doorOpen = false;
        isWaiting = true;
        if (currentLevel > 0 && currentLevel % 5 === 0 && hp < 3) hp++;
        generateMap();
        updateUI();
    }

    function updateUI() {
        document.getElementById('q-level').innerText = `STEP ${currentLevel + 1} / ${quizBank.length}`;
        document.getElementById('q-text').innerText = quizBank[currentLevel].q;
        document.getElementById('question-box').style.borderColor = "#4facfe";
        document.getElementById('status-hp').innerText = "HP: " + "❤️".repeat(hp);
        document.getElementById('status-score').innerText = "SCORE: " + score;
    }

    function showReviewScreen(title) {
        gameStarted = false;
        document.getElementById('review-screen').style.display = 'flex';
        document.getElementById('review-title').innerText = title;
        document.getElementById('final-score').innerText = "最終得分: " + score;
        const listDiv = document.getElementById('review-list');
        listDiv.innerHTML = wrongAnswers.length === 0 ? "<p>完美通關，幻境守護者！</p>" : 
            wrongAnswers.map(item => `<div class="review-item"><strong>問：${item.q}</strong><br><span style="color: #00f2fe;">正確答案：${item.a}</span></div>`).join('');
    }

    function loseHp() {
        const currentQ = quizBank[currentLevel];
        if (!wrongAnswers.find(item => item.q === currentQ.q)) {
            wrongAnswers.push({ q: currentQ.q, a: currentQ.options[currentQ.correct] });
        }
        hp--;
        if (hp <= 0) showReviewScreen("生命耗盡");
        else {
            updateUI();
            player.x = 50; player.y = 50;
            player.vx = 0; player.vy = 0;
            isWaiting = true;
        }
    }

    window.addEventListener('keydown', e => { 
        if(gameStarted) {
            if(isWaiting && e.code === 'Space') isWaiting = false;
            keys[e.code] = true; 
        }
    });
    window.addEventListener('keyup', e => { if(gameStarted) keys[e.code] = false; });

    function update() {
        if (!gameStarted || isWaiting) return;

        platforms.forEach(p => {
            if (p.isMovable) {
                p.x += bridgeDir * 1.5;
                if (p.x > p.maxX || p.x < p.minX) bridgeDir *= -1;
                if (player.onGround && player.y + player.size === p.y) player.x += bridgeDir * 1.5;
            }
        });

        coins.forEach(c => { c.x = platforms[c.pIdx].x + c.offsetX; });

        if (keys['ArrowLeft']) player.vx = -player.speed;
        else if (keys['ArrowRight']) player.vx = player.speed;
        else player.vx *= 0.8;

        if (keys['ArrowUp'] && player.onGround) { player.vy = player.jumpPower; player.onGround = false; }
        player.vy += player.gravity;
        player.x += player.vx;
        player.y += player.vy;
        player.onGround = false;

        platforms.forEach(p => {
            if (player.x + player.size > p.x && player.x < p.x + p.w &&
                player.y + player.size > p.y && player.y + player.size < p.y + p.h + 10 && player.vy >= 0) {
                player.y = p.y - player.size; player.vy = 0; player.onGround = true;
            }
        });

        coins.forEach(c => {
            if (!c.collected) {
                let range = 35;
                if (Math.abs(player.x - c.x) < range && Math.abs(player.y - c.y) < range) {
                    c.collected = true;
                    if (c.isCorrect) {
                        doorOpen = true; score += 100;
                        document.getElementById('q-text').innerText = "幻象破除！快去開啟傳送門！";
                        document.getElementById('question-box').style.backgroundColor = "rgba(46, 204, 113, 0.4)";
                        coins = coins.filter(coin => coin.isCorrect); 
                    } else { loseHp(); }
                }
            }
        });

        if (player.y > 520) loseHp();
        if (player.x < 0) player.x = 0;
        if (player.x > 800 - player.size) player.x = 800 - player.size;

        if (doorOpen && player.x > 710 && player.y > 360) {
            currentLevel++;
            if (currentLevel < quizBank.length) initLevel();
            else showReviewScreen("恭喜征服幻境！");
        }
        
        // 星星移動
        stars.forEach(s => {
            s.y += s.speed;
            if(s.y > 520) s.y = 0;
        });
    }

    function draw() {
        // 繪製背景漸層
        let grad = ctx.createLinearGradient(0, 0, 0, 520);
        grad.addColorStop(0, "#0f0c29");
        grad.addColorStop(1, "#24243e");
        ctx.fillStyle = grad;
        ctx.fillRect(0, 0, 800, 520);

        // 繪製星星
        ctx.fillStyle = "white";
        stars.forEach(s => {
            ctx.globalAlpha = Math.random(); // 閃爍感
            ctx.fillRect(s.x, s.y, s.size, s.size);
        });
        ctx.globalAlpha = 1.0;

        // 繪製發光平台
        platforms.forEach(p => { 
            ctx.shadowBlur = 10;
            ctx.shadowColor = p.isMovable ? "#e67e22" : "#4facfe";
            ctx.fillStyle = p.isMovable ? "#e67e22" : "#1a1a2e";
            ctx.fillRect(p.x, p.y, p.w, p.h);
            ctx.strokeStyle = p.isMovable ? "#f39c12" : "#4facfe";
            ctx.strokeRect(p.x, p.y, p.w, p.h);
        });
        ctx.shadowBlur = 0;

        // 繪製神祕金幣
        coins.forEach(c => {
            if (!c.collected) {
                ctx.shadowBlur = 15;
                ctx.shadowColor = "#f1c40f";
                ctx.fillStyle = "#f1c40f";
                ctx.beginPath(); ctx.arc(c.x+15, c.y+15, 14, 0, Math.PI*2); ctx.fill();
                ctx.shadowBlur = 0;
                ctx.fillStyle = "white"; ctx.font = "bold 15px Arial"; ctx.textAlign = "center";
                ctx.fillText(c.text, c.x + 15, c.y - 12);
            }
        });

        // 繪製傳送門
        if (doorOpen) {
            ctx.shadowBlur = 20;
            ctx.shadowColor = "#00f2fe";
            ctx.fillStyle = (Date.now() % 400 > 200) ? "#00f2fe" : "#4facfe";
            ctx.fillRect(720, 370, 50, 70);
        } else {
            ctx.fillStyle = "rgba(255, 255, 255, 0.1)";
            ctx.fillRect(720, 370, 50, 70);
            ctx.strokeStyle = "rgba(0, 242, 254, 0.3)";
            ctx.strokeRect(720, 370, 50, 70);
        }
        ctx.shadowBlur = 0;

        // 繪製主角 (帶外框發光)
        ctx.shadowBlur = 10;
        ctx.shadowColor = "#4ea5ff";
        ctx.fillStyle = "#4ea5ff";
        ctx.fillRect(player.x, player.y, player.size, player.size);
        ctx.strokeStyle = "white";
        ctx.strokeRect(player.x, player.y, player.size, player.size);
        ctx.shadowBlur = 0;

        if (isWaiting) {
            ctx.fillStyle = "rgba(0, 0, 0, 0.7)";
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = "#00f2fe";
            ctx.font = "bold 40px '微軟正黑體'";
            ctx.textAlign = "center";
            ctx.fillText("READY?", canvas.width / 2, canvas.height / 2);
            ctx.font = "20px '微軟正黑體'";
            ctx.fillStyle = "white";
            ctx.fillText("按下 [空白鍵] 或 [手機按鈕] 啟程", canvas.width / 2, canvas.height / 2 + 50);
        }
    }

    function gameLoop() { 
        if (gameStarted) { 
            update(); 
            draw(); 
            animationId = requestAnimationFrame(gameLoop); 
        } 
    }

    // 手機控制邏輯補強
    const btnMap = { 'btn-left': 'ArrowLeft', 'btn-right': 'ArrowRight', 'btn-jump': 'ArrowUp' };
    Object.keys(btnMap).forEach(id => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { 
            e.preventDefault(); 
            if(gameStarted) {
                if(isWaiting) isWaiting = false;
                keys[btnMap[id]] = true; 
            }
        });
        el.addEventListener('touchend', (e) => { 
            e.preventDefault(); 
            if(gameStarted) keys[btnMap[id]] = false; 
        });
    });
</script>
